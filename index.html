# -*- coding: utf-8 -*- import os from flask import Flask, request, render_template_string app = Flask(__name__) # مجلد لحفظ ملفات الـ RAW UPLOAD_FOLDER = 'raw_files' os.makedirs(UPLOAD_FOLDER, exist_ok=True) # واجهة الـ HTML مدمجة مباشرة داخل كود البايثون لسهولة التشغيل HTML_TEMPLATE = ''' TaTo RAW | لوحة تحكم السكربتات

TaTo RAW System

اكتب سكربت (Lua / Python) الخاص بك هنا للحصول على رابط RAW سريع لدلتا

TaTo Family - Owned by FAYO
''' # الصفحة الرئيسية تعرض واجهة التحكم (HTML) @app.route('/') def home(): return render_template_string(HTML_TEMPLATE) # استقبال البيانات وحفظها وتوليد رابط الـ RAW @app.route('/upload', methods=['POST']) def upload_file(): code_content = request.form.get('code') file_name = request.form.get('filename', 'script.lua') if not code_content: return "خطأ: لم يتم كتابة أي كود!", 400 # حفظ الملف داخل المجلد الخاص بالـ RAW file_path = os.path.join(UPLOAD_FOLDER, file_name) with open(file_path, 'w', encoding='utf-8') as f: f.write(code_content) # إرجاع رابط الـ RAW المباشر raw_url = f"{request.host_url}raw/{file_name}" # واجهة نجاح أنيقة تعرض الرابط success_html = f''' تم التوليد بنجاح!

تم رفع السكربت وتوليد الرابط بنجاح!

رابط RAW المباشر (جاهز للاستخدام في Delta أو بايثون):

{raw_url}


''' return render_template_string(success_html) # قراءة الملف كـ RAW (نص خام) @app.route('/raw/', methods=['GET']) def get_raw_file(filename): file_path = os.path.join(UPLOAD_FOLDER, filename) if os.path.exists(file_path): with open(file_path, 'r', encoding='utf-8') as f: return f.read(), 200, {'Content-Type': 'text/plain; charset=utf-8'} return "الملف غير موجود!", 404 if __name__ == '__main__': # تشغيل السيرفر على الشبكة المحلية ليعمل على أي جهاز متصل print("سيرفر TaTo RAW يعمل الآن!") app.run(host='0.0.0.0', port=5000, debug=True)
